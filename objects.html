Updation of captial by state dynamically


    arr = [{
        "state": "Telangana", "capital": "Hyderabad"
    }, {
        "state": "Andhra pradesh", "capital": "Amaravati"
    }]

    inputState = "Telangana";

    for (var key in arr) {
        if (arr[key].state == inputState) {
            var updState = {
                "state": arr[key].state,
                "capital": arr[key].capital
            };
        }
    }
    console.log(updState);
